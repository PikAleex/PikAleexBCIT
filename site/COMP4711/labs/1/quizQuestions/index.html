5) What is the resulting array of this script?

var hiWorld = (function() {
    var result = 1;
    for (let i = 2; i < 5; ++i) {
        result *= i;
    var greeting;
    result < 25 ? greeting = "Hi world!" : greeting = "Hello world!";
    var answer = [greeting, result];
    return answer;
})()

a) ["Hi world!", 24]
b) ["Hi world!", 23]
c) ["Hello world!", 25]
d) ["Hello world!", 26]
